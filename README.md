# kilikili
## config.ini
```
# debug开发模式,release生产模式
[service]
AppMode = debug
HttpPort = :3000
# 运行端口号 3000端口

[mysql]
Db = mysql
DbHost = 
# mysql ip地址
DbPort = 
# mysql 端口号
DbUser = 
# mysql 用户名
DbPassWord = 
# mysql 密码
DbName = 
# mysql 名字

[redis]
RedisDb = redis
RedisAddr = 
# redis ip地址和端口号
RedisPw =
# redis 密码
RedisDbName = 
# redis 名字
```
